                        class Calculator {
                            constructor(previousOperandTextElement, currentOperandTextElement) {
                              this.previousOperandTextElement = previousOperandTextElement;
                              this.currentOperandTextElement = currentOperandTextElement;
                              this.operations = {
                                add: "+",
                                substract: "-",
                                multiply: "*",
                                divide: "/",
                                exponentiation: "x\u207F",
                              };
                              this.clear();
                            }
                          
                            clear() {
                              this.currentOperand = "";
                              this.previousOperand = "";
                              this.operation = "";
                              this.toReset = false;
                            }
                          
                            delete() {
                              this.currentOperand = this.currentOperand.slice(0, -1);
                              if (this.currentOperand === "-") {
                                this.currentOperand = "";
                              }
                            }
                          
                            addDigit(digit) {
                              if (this.toReset) {
                                this.clear();
                              }
                          
                              if (digit === ".") {
                                if (this.currentOperand.includes(".")) {
                                  return;
                                }
                          
                                this.currentOperand = `${this.currentOperand || "0"}.`;
                                return;
                              }
                          
                              this.currentOperand = `${
                                this.currentOperand === "0" ? "" : this.currentOperand
                              }${digit}`;
                            }
                          
                            chooseOperationFunction(operation) {
                              if (this.currentOperand === "") {
                                return;
                              }
                          
                              if (operation === "negate") {
                                if (this.currentOperand !== "0") {
                                  if (this.currentOperand[0] === "-") {
                                    this.currentOperand = this.currentOperand.slice(1);
                                  } else {
                                    this.currentOperand = `-${this.currentOperand}`;
                                  }
                                }
                                return;
                              }
                          
                              const isEqualsOperation = operation === "equals";
                          
                              if (this.previousOperand !== "") {
                                this.calculate();
                          
                                if (isEqualsOperation) {
                                  this.toReset = true;
                                  return;
                                }
                              } else {
                                if (isEqualsOperation) {
                                  return;
                                }
                              }
                          
                              if (operation === "squareroot") {
                                if (this.currentOperand < 0) {
                                  this.currentOperand = "error";
                                } else {
                                  this.currentOperand = `${Math.sqrt(this.currentOperand)}`;
                                }
                                this.toReset = true;
                                return;
                              }
                          
                              this.operation = operation;
                              if (Number.isInteger(+this.currentOperand)) {
                                this.previousOperand = Math.trunc(this.currentOperand);
                              } else {
                                this.previousOperand = this.currentOperand;
                              }
                              this.currentOperand = "";
                              this.toReset = false;
                            }
                          
                            calculate() {
                              let calculation;
                              const previous = parseFloat(this.previousOperand);
                              const current = parseFloat(this.currentOperand);
                              if (isNaN(previous) || isNaN(current)) {
                                return;
                              }
                          
                              switch (this.operation) {
                                case "exponentiation":
                                  calculation = previous ** current;
                                  break;
                                case "add":
                                  calculation = previous + current;
                                  break;
                          
                                case "substract":
                                  calculation = previous - current;
                          
                                  break;
                          
                                case "multiply":
                                  calculation = previous * current;
                                  break;
                          
                                case "divide":
                                  calculation = previous / current;
                                  break;
                          
                                default:
                                  return;
                              }
                          
                              this.currentOperand = `${+calculation.toPrecision(8)}`;
                              this.operation = "";
                              this.previousOperand = "";
                            }
                          
                            updateOutput() {
                              this.currentOperandTextElement.innerText = this.currentOperand;
                              this.previousOperandTextElement.innerText = this.previousOperand;
                              if (this.operation) {
                                this.previousOperandTextElement.innerText = `${this.previousOperand} ${
                                  this.operations[this.operation]
                                }`;
                              }
                            }
                          }
                          
                          const digitsBtn = document.querySelectorAll("[data-digit]");
                          const operationsBtn = document.querySelectorAll("[data-operation]");
                          const clearBtn = document.querySelector("[data-all-clear]");
                          const deleteBtn = document.querySelector("[data-delete]");
                          const currentOperandTextElement = document.querySelector(
                            "[data-current-operand]"
                          );
                          const previousOperandTextElement = document.querySelector(
                            "[data-previous-operand]"
                          );
                          
                          const calculator = new Calculator(
                            previousOperandTextElement,
                            currentOperandTextElement
                          );
                          
                          digitsBtn.forEach((button) => {
                            button.addEventListener("click", () => {
                              calculator.addDigit(button.innerText);
                              calculator.updateOutput();
                            });
                          });
                          
                          deleteBtn.addEventListener("click", () => {
                            calculator.delete();
                            calculator.updateOutput();
                          });
                          
                          clearBtn.addEventListener("click", () => {
                            calculator.clear();
                            calculator.updateOutput();
                          });
                          
                          operationsBtn.forEach((button) => {
                            button.addEventListener("click", () => {
                              calculator.chooseOperationFunction(button.dataset.operation);
                              calculator.updateOutput();
                            });
                          });
